# з«ҷзӮ№еӨҡиҜӯиЁҖй…ҚзҪ®

иҰҒеҗҜз”Ё VuePress зҡ„еӨҡиҜӯиЁҖж”ҜжҢҒпјҢйҰ–е…ҲйңҖиҰҒдҪҝз”ЁеҰӮдёӢзҡ„ж–Үд»¶з»“жһ„пјҡ

docs
в”ңв”Җ README.md
в”ңв”Җ foo.md
в”ңв”Җ nested
в”ӮВ В в””в”Җ README.md
в””в”Җ zh
   в”ңв”Җ README.md
   в”ңв”Җ foo.md
   в””в”Җ nested
  В В   в””в”Җ README.md

з„¶еҗҺпјҢеңЁ .vuepress/config.js дёӯжҸҗдҫӣ locales йҖүйЎ№пјҡ

module.exports = {
  locales: {
    // й”®еҗҚжҳҜиҜҘиҜӯиЁҖжүҖеұһзҡ„еӯҗи·Ҝеҫ„
    // дҪңдёәзү№дҫӢпјҢй»ҳи®ӨиҜӯиЁҖеҸҜд»ҘдҪҝз”Ё '/' дҪңдёәе…¶и·Ҝеҫ„гҖӮ
    '/': {
      lang: 'en-US', // е°Ҷдјҡиў«и®ҫзҪ®дёә <html> зҡ„ lang еұһжҖ§
      title: 'VuePress',
      description: 'Vue-powered Static Site Generator'
    },
    '/zh/': {
      lang: 'zh-CN',
      title: 'VuePress',
      description: 'Vue й©ұеҠЁзҡ„йқҷжҖҒзҪ‘з«ҷз”ҹжҲҗеҷЁ'
    }
  }
}

еҰӮжһңдёҖдёӘиҜӯиЁҖжІЎжңүеЈ°жҳҺ title жҲ–иҖ… descriptionпјҢVuePress е°Ҷдјҡе°қиҜ•дҪҝз”Ёй…ҚзҪ®йЎ¶еұӮзҡ„еҜ№еә”еҖјгҖӮеҰӮжһңжҜҸдёӘиҜӯиЁҖйғҪеЈ°жҳҺдәҶ title е’Ң descriptionпјҢеҲҷйЎ¶еұӮзҡ„иҝҷдёӨдёӘеҖјеҸҜд»Ҙиў«зңҒз•ҘгҖӮ

# й»ҳи®Өдё»йўҳеӨҡиҜӯиЁҖй…ҚзҪ®

й»ҳи®Өдё»йўҳд№ҹеҶ…зҪ®дәҶеӨҡиҜӯиЁҖж”ҜжҢҒпјҢеҸҜд»ҘйҖҡиҝҮ themeConfig.locales жқҘй…ҚзҪ®гҖӮиҜҘйҖүйЎ№жҺҘеҸ—еҗҢж ·зҡ„ { path: config } ж јејҸзҡ„еҖјгҖӮжҜҸдёӘиҜӯиЁҖйҷӨдәҶеҸҜд»Ҙй…ҚзҪ®дёҖдәӣз«ҷзӮ№дёӯз”ЁеҲ°зҡ„ж–Үеӯ—д№ӢеӨ–пјҢиҝҳеҸҜд»ҘжӢҘжңүиҮӘе·ұзҡ„ еҜјиҲӘж Ҹ е’Ң дҫ§иҫ№ж Ҹ й…ҚзҪ®пјҡ

module.exports = {
  locales: { /* ... */ },
  themeConfig: {
    locales: {
      '/': {
        selectText: 'Languages',
        label: 'English',
        editLinkText: 'Edit this page on GitHub',
        serviceWorker: {
          updatePopup: {
            message: "New content is available.",
            buttonText: "Refresh"
          }
        },
        algolia: {},
        nav: [
          { text: 'Nested', link: '/nested/' }
        ],
        sidebar: {
          '/': [/* ... */],
          '/nested/': [/* ... */]
        }
      },
      '/zh/': {
        // еӨҡиҜӯиЁҖдёӢжӢүиҸңеҚ•зҡ„ж Үйўҳ
        selectText: 'йҖүжӢ©иҜӯиЁҖ',
        // иҜҘиҜӯиЁҖеңЁдёӢжӢүиҸңеҚ•дёӯзҡ„ж Үзӯҫ
        label: 'з®ҖдҪ“дёӯж–Ү',
        // зј–иҫ‘й“ҫжҺҘж–Үеӯ—
        editLinkText: 'еңЁ GitHub дёҠзј–иҫ‘жӯӨйЎө',
        // Service Worker зҡ„й…ҚзҪ®
        serviceWorker: {
          updatePopup: {
            message: "еҸ‘зҺ°ж–°еҶ…е®№еҸҜз”Ё.",
            buttonText: "еҲ·ж–°"
          }
        },
        // еҪ“еүҚ locale зҡ„ algolia docsearch йҖүйЎ№
        algolia: {},
        nav: [
          { text: 'еөҢеҘ—', link: '/zh/nested/' }
        ],
        sidebar: {
          '/zh/': [/* ... */],
          '/zh/nested/': [/* ... */]
        }
      }
    }
  }
}
